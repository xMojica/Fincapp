
-- Eliminar las tablas si existen
DROP TABLE IF EXISTS engorde;
DROP TABLE IF EXISTS ordeño;
DROP TABLE IF EXISTS genetica;
DROP TABLE IF EXISTS finanzas;
DROP TABLE IF EXISTS persona_animales;
DROP TABLE IF EXISTS venta;
DROP TABLE IF EXISTS persona;
DROP TABLE IF EXISTS animal;
DROP TABLE IF EXISTS corral;

-- Crear la tabla animal
CREATE TABLE animal (
    id               VARCHAR(5) NOT NULL,
    raza             VARCHAR(20) NOT NULL,
    peso             FLOAT NOT NULL,
    sexo             VARCHAR(1) NOT NULL,
    color            VARCHAR(20) NOT NULL,
    fecha_nacimiento DATE NOT NULL,
    fecha_ingreso    DATE NOT NULL,
    precio_compra    INTEGER NOT NULL,
    corral_id        VARCHAR(20) NOT NULL,
    edad_actual      INTEGER NOT NULL,
    edad_compra      INTEGER NOT NULL
);

-- Definir la clave primaria para la tabla animal
ALTER TABLE animal ADD CONSTRAINT animal_pk PRIMARY KEY (id);

-- Crear la tabla corral
CREATE TABLE corral (
    id        VARCHAR(20) NOT NULL,
    capacidad INTEGER NOT NULL,
    dimension VARCHAR(20) NOT NULL
);

-- Definir la clave primaria para la tabla corral
ALTER TABLE corral ADD CONSTRAINT corral_pk PRIMARY KEY (id);

-- Crear la tabla engorde
CREATE TABLE engorde (
    id           VARCHAR(5) NOT NULL,
    peso_ingreso FLOAT NOT NULL,
    precio_kilo INTEGER NOT NULL
);

-- Definir la clave primaria para la tabla engorde
ALTER TABLE engorde ADD CONSTRAINT engorde_pk PRIMARY KEY (id);

-- Crear la tabla finanzas
CREATE TABLE finanzas (
    gastos_personal                  FLOAT NOT NULL,
    gastos_animales                  FLOAT NOT NULL,
    gastos_transporte                FLOAT NOT NULL,
    gastos_estructurales             FLOAT NOT NULL,
    saldo                            FLOAT NOT NULL,
    porcentaje_rentabilidad_absoluto INTEGER NOT NULL
);

-- Crear la tabla genetica
CREATE TABLE genetica (
    id                VARCHAR(5) NOT NULL,
    porcentaje_pureza INTEGER NOT NULL,
    raza_padre        VARCHAR(20) NOT NULL,
    raza_madre        VARCHAR(20) NOT NULL
);

-- Definir la clave primaria para la tabla genetica
ALTER TABLE genetica ADD CONSTRAINT genetica_pk PRIMARY KEY (id);

-- Crear la tabla ordeño
CREATE TABLE ordeño (
    id           VARCHAR(5) NOT NULL,
    cantidad     FLOAT NOT NULL,
    precio_litro INTEGER NOT NULL
);

-- Definir la clave primaria para la tabla ordeño
ALTER TABLE ordeño ADD CONSTRAINT ordeño_pk PRIMARY KEY (id);

-- Crear la tabla persona
CREATE TABLE persona (
    id       VARCHAR(20) NOT NULL,
    nombre   VARCHAR(20) NOT NULL,
    apellido VARCHAR(20) NOT NULL,
    contacto VARCHAR(10) NOT NULL,
    area     VARCHAR(20) NOT NULL
);

-- Definir la clave primaria para la tabla persona
ALTER TABLE persona ADD CONSTRAINT persona_pk PRIMARY KEY (id);

-- Crear la tabla persona_animales
CREATE TABLE persona_animales (
    persona_id VARCHAR(20) NOT NULL,
    animal_id  VARCHAR(5) NOT NULL
);

-- Definir la clave primaria para la tabla persona_animales
ALTER TABLE persona_animales ADD CONSTRAINT persona_animales_pk PRIMARY KEY (persona_id, animal_id);

-- Crear la tabla venta
CREATE TABLE venta (
    id                    VARCHAR(20) NOT NULL,
    fecha_venta           DATE NOT NULL,
    precio_venta          FLOAT NOT NULL,
    rentabilidad_relativa FLOAT NOT NULL,
    animal_id             VARCHAR(5) NOT NULL
);

-- Crear un índice único para la columna animal_id en la tabla venta
CREATE UNIQUE INDEX sale_idx ON venta (animal_id);

-- Definir la clave primaria para la tabla venta
ALTER TABLE venta ADD CONSTRAINT venta_pk PRIMARY KEY (id);

ALTER TABLE persona_animales
